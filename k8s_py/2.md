## 1. Обновленный `collect_inventory.py`

Я добавил параметры:

* `--exclude-label key=value` — можно указывать несколько раз
* `--exclude-label-key key` — исключать ноды, у которых есть этот label key
* `--exclude-taint-key key` — исключать ноды, у которых есть этот taint key

Это удобно, потому что ingress-ноды в разных кластерах помечаются по-разному.

Примеры:

* `ingress=true`
* `node-role.kubernetes.io/ingress`
* taint `node-role.kubernetes.io/ingress:NoSchedule`

```python id="cxdr0p"
#!/usr/bin/env python3
import argparse
import json
import subprocess
from decimal import Decimal
from collections import defaultdict


def run_kubectl(args):
    cmd = ["kubectl"] + args
    result = subprocess.run(cmd, capture_output=True, text=True, check=True)
    return result.stdout


def parse_cpu(cpu_str):
    cpu_str = str(cpu_str).strip()
    if not cpu_str:
        return 0.0
    if cpu_str.endswith("m"):
        return float(Decimal(cpu_str[:-1]) / Decimal(1000))
    return float(Decimal(cpu_str))


def parse_memory(mem_str):
    mem_str = str(mem_str).strip()
    if not mem_str:
        return 0.0

    binary_units = {
        "Ki": 1024,
        "Mi": 1024 ** 2,
        "Gi": 1024 ** 3,
        "Ti": 1024 ** 4,
        "Pi": 1024 ** 5,
        "Ei": 1024 ** 6,
    }
    decimal_units = {
        "K": 1000,
        "M": 1000 ** 2,
        "G": 1000 ** 3,
        "T": 1000 ** 4,
        "P": 1000 ** 5,
        "E": 1000 ** 6,
    }

    for unit, factor in binary_units.items():
        if mem_str.endswith(unit):
            val = Decimal(mem_str[:-len(unit)])
            return float(val * factor / Decimal(1024 ** 3))

    for unit, factor in decimal_units.items():
        if mem_str.endswith(unit):
            val = Decimal(mem_str[:-len(unit)])
            return float(val * factor / Decimal(1024 ** 3))

    return float(Decimal(mem_str) / Decimal(1024 ** 3))


def owner_ref_kind_and_name(owner_refs):
    if not owner_refs:
        return ("Pod", "standalone")

    ref = owner_refs[0]
    kind = ref.get("kind", "Unknown")
    name = ref.get("name", "unknown")

    if kind == "ReplicaSet":
        parts = name.rsplit("-", 1)
        if len(parts) == 2 and parts[1]:
            return ("Deployment", parts[0])
        return ("ReplicaSet", name)

    return (kind, name)


def parse_key_value(items):
    result = []
    for item in items:
        if "=" not in item:
            raise ValueError(f"Expected key=value, got: {item}")
        key, value = item.split("=", 1)
        result.append((key.strip(), value.strip()))
    return result


def node_matches_exclude(labels, taints, exclude_labels, exclude_label_keys, exclude_taint_keys):
    for key, value in exclude_labels:
        if labels.get(key) == value:
            return True

    for key in exclude_label_keys:
        if key in labels:
            return True

    taint_keys = {t.get("key") for t in taints if "key" in t}
    for key in exclude_taint_keys:
        if key in taint_keys:
            return True

    return False


def get_worker_nodes(exclude_labels, exclude_label_keys, exclude_taint_keys):
    raw = run_kubectl(["get", "nodes", "-o", "json"])
    data = json.loads(raw)

    workers = []
    excluded_nodes = []

    for item in data["items"]:
        name = item["metadata"]["name"]
        labels = item["metadata"].get("labels", {})
        taints = item["spec"].get("taints", [])

        is_control_plane = (
            "node-role.kubernetes.io/control-plane" in labels
            or "node-role.kubernetes.io/master" in labels
        )
        if is_control_plane:
            excluded_nodes.append({"name": name, "reason": "control-plane"})
            continue

        if node_matches_exclude(labels, taints, exclude_labels, exclude_label_keys, exclude_taint_keys):
            excluded_nodes.append({"name": name, "reason": "matched_exclude_rule"})
            continue

        unschedulable = item["spec"].get("unschedulable", False)

        alloc = item["status"]["allocatable"]
        cpu = parse_cpu(alloc.get("cpu", "0"))
        mem_gib = parse_memory(alloc.get("memory", "0"))

        workers.append({
            "name": name,
            "allocatable_cpu_cores": cpu,
            "allocatable_mem_gib": mem_gib,
            "unschedulable": unschedulable,
            "taints": taints,
            "labels": labels,
        })

    return workers, excluded_nodes


def sum_container_resources(containers):
    req_cpu = 0.0
    req_mem = 0.0
    lim_cpu = 0.0
    lim_mem = 0.0

    for c in containers:
        resources = c.get("resources", {})
        requests = resources.get("requests", {})
        limits = resources.get("limits", {})

        req_cpu += parse_cpu(requests.get("cpu", "0"))
        req_mem += parse_memory(requests.get("memory", "0"))
        lim_cpu += parse_cpu(limits.get("cpu", "0"))
        lim_mem += parse_memory(limits.get("memory", "0"))

    return {
        "requests_cpu_cores": req_cpu,
        "requests_mem_gib": req_mem,
        "limits_cpu_cores": lim_cpu,
        "limits_mem_gib": lim_mem,
    }


def pod_effective_resources(spec):
    containers = spec.get("containers", [])
    init_containers = spec.get("initContainers", [])

    normal = sum_container_resources(containers)

    init_values = [sum_container_resources([c]) for c in init_containers]
    init_req_cpu = max((x["requests_cpu_cores"] for x in init_values), default=0.0)
    init_req_mem = max((x["requests_mem_gib"] for x in init_values), default=0.0)
    init_lim_cpu = max((x["limits_cpu_cores"] for x in init_values), default=0.0)
    init_lim_mem = max((x["limits_mem_gib"] for x in init_values), default=0.0)

    return {
        "requests_cpu_cores": max(normal["requests_cpu_cores"], init_req_cpu),
        "requests_mem_gib": max(normal["requests_mem_gib"], init_req_mem),
        "limits_cpu_cores": max(normal["limits_cpu_cores"], init_lim_cpu),
        "limits_mem_gib": max(normal["limits_mem_gib"], init_lim_mem),
    }


def empty_resource_totals():
    return {
        "requests_cpu_cores": 0.0,
        "requests_mem_gib": 0.0,
        "limits_cpu_cores": 0.0,
        "limits_mem_gib": 0.0,
        "pod_count": 0,
    }


def add_resource_totals(dst, src):
    dst["requests_cpu_cores"] += src["requests_cpu_cores"]
    dst["requests_mem_gib"] += src["requests_mem_gib"]
    dst["limits_cpu_cores"] += src["limits_cpu_cores"]
    dst["limits_mem_gib"] += src["limits_mem_gib"]
    dst["pod_count"] += 1


def get_pods():
    raw = run_kubectl(["get", "pods", "-A", "-o", "json"])
    data = json.loads(raw)

    pods = []

    regular_totals = empty_resource_totals()
    daemonset_totals = empty_resource_totals()

    namespace_summary = defaultdict(empty_resource_totals)
    workload_summary = defaultdict(empty_resource_totals)
    daemonset_per_pod_summary = defaultdict(empty_resource_totals)

    for item in data["items"]:
        meta = item.get("metadata", {})
        spec = item.get("spec", {})
        status = item.get("status", {})

        phase = status.get("phase", "")
        if phase not in {"Running", "Pending"}:
            continue

        namespace = meta.get("namespace", "")
        name = meta.get("name", "")
        node_name = spec.get("nodeName")
        owner_refs = meta.get("ownerReferences", [])

        workload_kind, workload_name = owner_ref_kind_and_name(owner_refs)
        resources = pod_effective_resources(spec)

        pod_info = {
            "namespace": namespace,
            "name": name,
            "node_name": node_name,
            "phase": phase,
            "workload_kind": workload_kind,
            "workload_name": workload_name,
            "is_daemonset": workload_kind == "DaemonSet",
            **resources,
        }
        pods.append(pod_info)

        ns_key = namespace
        wl_key = f"{namespace}/{workload_kind}/{workload_name}"

        add_resource_totals(namespace_summary[ns_key], resources)
        add_resource_totals(workload_summary[wl_key], resources)

        if workload_kind == "DaemonSet":
            add_resource_totals(daemonset_totals, resources)
            add_resource_totals(daemonset_per_pod_summary[wl_key], resources)
        else:
            add_resource_totals(regular_totals, resources)

    daemonset_per_pod_list = []
    for key, val in sorted(daemonset_per_pod_summary.items()):
        namespace, kind, name = key.split("/", 2)
        daemonset_per_pod_list.append({
            "namespace": namespace,
            "workload_kind": kind,
            "workload_name": name,
            **val,
        })

    namespace_list = []
    for key, val in sorted(namespace_summary.items()):
        namespace_list.append({
            "namespace": key,
            **val,
        })

    workload_list = []
    for key, val in sorted(workload_summary.items()):
        namespace, kind, name = key.split("/", 2)
        workload_list.append({
            "namespace": namespace,
            "workload_kind": kind,
            "workload_name": name,
            **val,
        })

    return {
        "pods": pods,
        "regular_totals": regular_totals,
        "daemonset_totals_current_cluster": daemonset_totals,
        "namespace_summary": namespace_list,
        "workload_summary": workload_list,
        "daemonset_per_pod_summary": daemonset_per_pod_list,
    }


def main():
    parser = argparse.ArgumentParser(description="Collect Kubernetes worker/node and pod capacity inventory")
    parser.add_argument("-o", "--output", default="inventory.json", help="Output JSON file")
    parser.add_argument(
        "--exclude-label",
        action="append",
        default=[],
        help="Exclude nodes by exact label match, may be repeated, format key=value"
    )
    parser.add_argument(
        "--exclude-label-key",
        action="append",
        default=[],
        help="Exclude nodes if label key exists, may be repeated"
    )
    parser.add_argument(
        "--exclude-taint-key",
        action="append",
        default=[],
        help="Exclude nodes if taint key exists, may be repeated"
    )
    args = parser.parse_args()

    exclude_labels = parse_key_value(args.exclude_label)

    workers, excluded_nodes = get_worker_nodes(
        exclude_labels=exclude_labels,
        exclude_label_keys=args.exclude_label_key,
        exclude_taint_keys=args.exclude_taint_key,
    )
    pod_data = get_pods()

    node_totals = {
        "allocatable_cpu_cores": sum(n["allocatable_cpu_cores"] for n in workers),
        "allocatable_mem_gib": sum(n["allocatable_mem_gib"] for n in workers),
        "worker_node_count": len(workers),
    }

    result = {
        "nodes": workers,
        "excluded_nodes": excluded_nodes,
        "exclude_rules": {
            "exclude_labels": [{"key": k, "value": v} for k, v in exclude_labels],
            "exclude_label_keys": args.exclude_label_key,
            "exclude_taint_keys": args.exclude_taint_key,
        },
        "node_totals": node_totals,
        "pods": pod_data["pods"],
        "regular_totals": pod_data["regular_totals"],
        "daemonset_totals_current_cluster": pod_data["daemonset_totals_current_cluster"],
        "namespace_summary": pod_data["namespace_summary"],
        "workload_summary": pod_data["workload_summary"],
        "daemonset_per_pod_summary": pod_data["daemonset_per_pod_summary"],
    }

    with open(args.output, "w", encoding="utf-8") as f:
        json.dump(result, f, indent=2)

    print(f"Saved inventory to {args.output}")
    print(json.dumps({
        "worker_nodes_after_exclude": node_totals["worker_node_count"],
        "excluded_nodes": len(excluded_nodes),
        "allocatable_cpu_cores": round(node_totals["allocatable_cpu_cores"], 3),
        "allocatable_mem_gib": round(node_totals["allocatable_mem_gib"], 3),
        "regular_requests_cpu_cores": round(result["regular_totals"]["requests_cpu_cores"], 3),
        "regular_requests_mem_gib": round(result["regular_totals"]["requests_mem_gib"], 3),
        "daemonset_requests_cpu_cores_current_cluster": round(result["daemonset_totals_current_cluster"]["requests_cpu_cores"], 3),
        "daemonset_requests_mem_gib_current_cluster": round(result["daemonset_totals_current_cluster"]["requests_mem_gib"], 3),
    }, indent=2))


if __name__ == "__main__":
    main()
```

```python id="m58i3u"
#!/usr/bin/env python3
import argparse
import json
from copy import deepcopy


def load_json(path):
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)


def capacity_for_nodes(nodes_by_name, node_names):
    selected = []
    for name in node_names:
        if name not in nodes_by_name:
            raise ValueError(f"Node '{name}' not found in inventory")
        selected.append(nodes_by_name[name])

    return {
        "nodes": selected,
        "allocatable_cpu_cores": sum(n["allocatable_cpu_cores"] for n in selected),
        "allocatable_mem_gib": sum(n["allocatable_mem_gib"] for n in selected),
        "node_count": len(selected),
        "largest_node": largest_node(selected),
    }


def largest_node(nodes):
    if not nodes:
        return None
    return max(nodes, key=lambda n: (n["allocatable_cpu_cores"], n["allocatable_mem_gib"]))


def compute_daemonset_overhead(inventory, node_count, basis):
    current_ds = inventory["daemonset_totals_current_cluster"]
    current_node_count = inventory["node_totals"]["worker_node_count"]

    if current_node_count <= 0:
        raise ValueError("Current cluster has zero worker nodes in inventory")

    per_node_cpu = current_ds[f"{basis}_cpu_cores"] / current_node_count
    per_node_mem = current_ds[f"{basis}_mem_gib"] / current_node_count

    return {
        "per_node_cpu_cores": per_node_cpu,
        "per_node_mem_gib": per_node_mem,
        "total_cpu_cores": per_node_cpu * node_count,
        "total_mem_gib": per_node_mem * node_count,
    }


def regular_required(inventory, basis, headroom_percent):
    regular = inventory["regular_totals"]
    mult = 1.0 + headroom_percent / 100.0
    return {
        "cpu_cores": regular[f"{basis}_cpu_cores"] * mult,
        "mem_gib": regular[f"{basis}_mem_gib"] * mult,
    }


def total_required_for_cluster(inventory, basis, node_count, headroom_percent):
    reg = regular_required(inventory, basis, headroom_percent)
    ds = compute_daemonset_overhead(inventory, node_count, basis)

    return {
        "regular_cpu_cores": reg["cpu_cores"],
        "regular_mem_gib": reg["mem_gib"],
        "daemonset_cpu_cores": ds["total_cpu_cores"],
        "daemonset_mem_gib": ds["total_mem_gib"],
        "total_cpu_cores": reg["cpu_cores"] + ds["total_cpu_cores"],
        "total_mem_gib": reg["mem_gib"] + ds["total_mem_gib"],
        "daemonset_per_node_cpu_cores": ds["per_node_cpu_cores"],
        "daemonset_per_node_mem_gib": ds["per_node_mem_gib"],
    }


def fit_status(cap_cpu, cap_mem, req_cpu, req_mem):
    cpu_ok = cap_cpu >= req_cpu
    mem_ok = cap_mem >= req_mem
    return {
        "cpu_ok": cpu_ok,
        "mem_ok": mem_ok,
        "all_ok": cpu_ok and mem_ok,
        "cpu_margin": cap_cpu - req_cpu,
        "mem_margin": cap_mem - req_mem,
        "blocking_resource": blocking_resource(cpu_ok, mem_ok),
        "cpu_ratio": (cap_cpu / req_cpu) if req_cpu > 0 else None,
        "mem_ratio": (cap_mem / req_mem) if req_mem > 0 else None,
    }


def blocking_resource(cpu_ok, mem_ok):
    if cpu_ok and mem_ok:
        return None
    if not cpu_ok and not mem_ok:
        return "cpu_and_memory"
    if not cpu_ok:
        return "cpu"
    return "memory"


def sortable_node_weight(node):
    return (node["allocatable_cpu_cores"], node["allocatable_mem_gib"])


def max_nodes_for_maintenance(cluster_nodes, inventory, basis, headroom_percent):
    remaining = sorted(deepcopy(cluster_nodes), key=sortable_node_weight)
    removed = []

    while remaining:
        node = remaining[0]
        candidate = remaining[1:]

        req = total_required_for_cluster(
            inventory=inventory,
            basis=basis,
            node_count=len(candidate),
            headroom_percent=headroom_percent,
        )

        cap_cpu = sum(n["allocatable_cpu_cores"] for n in candidate)
        cap_mem = sum(n["allocatable_mem_gib"] for n in candidate)
        status = fit_status(cap_cpu, cap_mem, req["total_cpu_cores"], req["total_mem_gib"])

        if status["all_ok"]:
            removed.append(node)
            remaining = candidate
        else:
            break

    final_req = total_required_for_cluster(
        inventory=inventory,
        basis=basis,
        node_count=len(remaining),
        headroom_percent=headroom_percent,
    )

    return {
        "max_maintainable_node_count": len(removed),
        "maintainable_nodes": [n["name"] for n in removed],
        "remaining_node_count": len(remaining),
        "remaining_capacity_cpu_cores": sum(n["allocatable_cpu_cores"] for n in remaining),
        "remaining_capacity_mem_gib": sum(n["allocatable_mem_gib"] for n in remaining),
        "remaining_required_cpu_cores": final_req["total_cpu_cores"],
        "remaining_required_mem_gib": final_req["total_mem_gib"],
    }


def survives_loss_of_any_one_node(cluster_nodes, inventory, basis, headroom_percent):
    failures = []

    total_cpu = sum(n["allocatable_cpu_cores"] for n in cluster_nodes)
    total_mem = sum(n["allocatable_mem_gib"] for n in cluster_nodes)

    for node in cluster_nodes:
        node_count_after = len(cluster_nodes) - 1
        req = total_required_for_cluster(
            inventory=inventory,
            basis=basis,
            node_count=node_count_after,
            headroom_percent=headroom_percent,
        )

        cpu_after = total_cpu - node["allocatable_cpu_cores"]
        mem_after = total_mem - node["allocatable_mem_gib"]
        status = fit_status(cpu_after, mem_after, req["total_cpu_cores"], req["total_mem_gib"])

        failures.append({
            "lost_node": node["name"],
            "capacity_after_loss_cpu_cores": cpu_after,
            "capacity_after_loss_mem_gib": mem_after,
            "required_after_loss_cpu_cores": req["total_cpu_cores"],
            "required_after_loss_mem_gib": req["total_mem_gib"],
            "cpu_ok": status["cpu_ok"],
            "mem_ok": status["mem_ok"],
            "all_ok": status["all_ok"],
            "blocking_resource": status["blocking_resource"],
            "cpu_margin": status["cpu_margin"],
            "mem_margin": status["mem_margin"],
        })

    all_ok = all(x["all_ok"] for x in failures)
    return {
        "survives_loss_of_any_one_node": all_ok,
        "details": failures,
    }


def survives_loss_of_largest_node(cluster_nodes, inventory, basis, headroom_percent):
    if not cluster_nodes:
        return {
            "survives_loss_of_largest_node": False,
            "reason": "empty_cluster",
        }

    biggest = largest_node(cluster_nodes)
    total_cpu = sum(n["allocatable_cpu_cores"] for n in cluster_nodes)
    total_mem = sum(n["allocatable_mem_gib"] for n in cluster_nodes)

    node_count_after = len(cluster_nodes) - 1
    req = total_required_for_cluster(
        inventory=inventory,
        basis=basis,
        node_count=node_count_after,
        headroom_percent=headroom_percent,
    )

    cpu_after = total_cpu - biggest["allocatable_cpu_cores"]
    mem_after = total_mem - biggest["allocatable_mem_gib"]
    status = fit_status(cpu_after, mem_after, req["total_cpu_cores"], req["total_mem_gib"])

    return {
        "survives_loss_of_largest_node": status["all_ok"],
        "largest_node": biggest["name"],
        "largest_node_cpu_cores": biggest["allocatable_cpu_cores"],
        "largest_node_mem_gib": biggest["allocatable_mem_gib"],
        "capacity_after_loss_cpu_cores": cpu_after,
        "capacity_after_loss_mem_gib": mem_after,
        "required_after_loss_cpu_cores": req["total_cpu_cores"],
        "required_after_loss_mem_gib": req["total_mem_gib"],
        "cpu_ok": status["cpu_ok"],
        "mem_ok": status["mem_ok"],
        "blocking_resource": status["blocking_resource"],
        "cpu_margin": status["cpu_margin"],
        "mem_margin": status["mem_margin"],
    }


def top_consumers(summary_list, basis, top_n):
    cpu_key = f"{basis}_cpu_cores"
    mem_key = f"{basis}_mem_gib"

    top_cpu = sorted(summary_list, key=lambda x: x[cpu_key], reverse=True)[:top_n]
    top_mem = sorted(summary_list, key=lambda x: x[mem_key], reverse=True)[:top_n]

    return {
        "top_cpu": top_cpu,
        "top_mem": top_mem,
    }


def summarize_cluster(name, cluster_node_names, nodes_by_name, inventory, basis, headroom_percent, top_n):
    cap = capacity_for_nodes(nodes_by_name, cluster_node_names)
    req = total_required_for_cluster(
        inventory=inventory,
        basis=basis,
        node_count=cap["node_count"],
        headroom_percent=headroom_percent,
    )

    status = fit_status(
        cap["allocatable_cpu_cores"],
        cap["allocatable_mem_gib"],
        req["total_cpu_cores"],
        req["total_mem_gib"],
    )

    maint = max_nodes_for_maintenance(
        cluster_nodes=cap["nodes"],
        inventory=inventory,
        basis=basis,
        headroom_percent=headroom_percent,
    )
    n_minus_1 = survives_loss_of_any_one_node(
        cluster_nodes=cap["nodes"],
        inventory=inventory,
        basis=basis,
        headroom_percent=headroom_percent,
    )
    largest_loss = survives_loss_of_largest_node(
        cluster_nodes=cap["nodes"],
        inventory=inventory,
        basis=basis,
        headroom_percent=headroom_percent,
    )

    return {
        "cluster": name,
        "node_count": cap["node_count"],
        "nodes": cluster_node_names,
        "largest_node": cap["largest_node"],
        "capacity": {
            "allocatable_cpu_cores": cap["allocatable_cpu_cores"],
            "allocatable_mem_gib": cap["allocatable_mem_gib"],
        },
        "required": req,
        "fit": status,
        "maintenance": maint,
        "n_minus_1": n_minus_1,
        "largest_node_failure": largest_loss,
        "imbalances": {
            "passes_cpu_but_fails_memory": status["cpu_ok"] and not status["mem_ok"],
            "passes_memory_but_fails_cpu": status["mem_ok"] and not status["cpu_ok"],
            "fails_both_cpu_and_memory": (not status["cpu_ok"]) and (not status["mem_ok"]),
        },
        "top_namespace_consumers": top_consumers(inventory["namespace_summary"], basis, top_n),
        "top_workload_consumers": top_consumers(inventory["workload_summary"], basis, top_n),
    }


def print_top_list(title, items, basis):
    cpu_key = f"{basis}_cpu_cores"
    mem_key = f"{basis}_mem_gib"

    print(title)
    if not items:
        print("  (none)")
        return

    for item in items:
        if "workload_kind" in item:
            ident = f"{item['namespace']}/{item['workload_kind']}/{item['workload_name']}"
        else:
            ident = item["namespace"]

        print(
            f"  {ident}: "
            f"pods={item['pod_count']}, "
            f"cpu={item[cpu_key]:.3f}, "
            f"mem={item[mem_key]:.3f} GiB"
        )

def main():
    parser = argparse.ArgumentParser(description="Evaluate split of worker nodes into two clusters")
    parser.add_argument("--inventory", required=True, help="Path to inventory.json")
    parser.add_argument("--split", required=True, help="Path to split.json")
    parser.add_argument(
        "--basis",
        choices=["requests", "limits"],
        default="requests",
        help="Which resource basis to use"
    )
    parser.add_argument(
        "--headroom-percent",
        type=float,
        default=0.0,
        help="Extra safety margin, e.g. 20 for +20%%"
    )
    parser.add_argument(
        "--top-n",
        type=int,
        default=10,
        help="How many top namespace/workload consumers to print"
    )
    parser.add_argument(
        "--brief",
        action="store_true",
        help="Print only brief summary without top namespace/workload sections"
    )
    args = parser.parse_args()

    inventory = load_json(args.inventory)
    split = load_json(args.split)

    nodes = inventory["nodes"]
    nodes_by_name = {n["name"]: n for n in nodes}

    all_inventory_nodes = set(nodes_by_name.keys())
    cluster_a_nodes = split.get("cluster_a", [])
    cluster_b_nodes = split.get("cluster_b", [])

    assigned = set(cluster_a_nodes) | set(cluster_b_nodes)
    duplicates = set(cluster_a_nodes) & set(cluster_b_nodes)
    unassigned = all_inventory_nodes - assigned
    unknown = assigned - all_inventory_nodes

    if duplicates:
        raise ValueError(f"Nodes assigned to both clusters: {sorted(duplicates)}")
    if unknown:
        raise ValueError(f"Unknown nodes in split file: {sorted(unknown)}")
    if unassigned:
        raise ValueError(f"Unassigned nodes: {sorted(unassigned)}")

    cluster_a = summarize_cluster(
        name="cluster_a",
        cluster_node_names=cluster_a_nodes,
        nodes_by_name=nodes_by_name,
        inventory=inventory,
        basis=args.basis,
        headroom_percent=args.headroom_percent,
        top_n=args.top_n,
    )

    cluster_b = summarize_cluster(
        name="cluster_b",
        cluster_node_names=cluster_b_nodes,
        nodes_by_name=nodes_by_name,
        inventory=inventory,
        basis=args.basis,
        headroom_percent=args.headroom_percent,
        top_n=args.top_n,
    )

    report = {
        "basis": args.basis,
        "headroom_percent": args.headroom_percent,
        "brief": args.brief,
        "current_cluster_worker_node_count": inventory["node_totals"]["worker_node_count"],
        "excluded_nodes": inventory.get("excluded_nodes", []),
        "exclude_rules": inventory.get("exclude_rules", {}),
        "current_cluster_regular_totals": inventory["regular_totals"],
        "current_cluster_daemonset_totals": inventory["daemonset_totals_current_cluster"],
        "cluster_a": cluster_a,
        "cluster_b": cluster_b,
    }

    print(json.dumps(report, indent=2))

    for cluster in [cluster_a, cluster_b]:
        print_cluster_summary(cluster)

        if not args.brief:
            print_top_list(
                f"top namespaces by CPU ({args.basis}):",
                cluster["top_namespace_consumers"]["top_cpu"],
                args.basis,
            )
            print_top_list(
                f"top namespaces by MEM ({args.basis}):",
                cluster["top_namespace_consumers"]["top_mem"],
                args.basis,
            )
            print_top_list(
                f"top workloads by CPU ({args.basis}):",
                cluster["top_workload_consumers"]["top_cpu"],
                args.basis,
            )
            print_top_list(
                f"top workloads by MEM ({args.basis}):",
                cluster["top_workload_consumers"]["top_mem"],
                args.basis,
            )


if __name__ == "__main__":
    main()
```

## Что важно учесть

Сейчас ingress-ноды **полностью исключаются из worker inventory**.
Это правильно, если они действительно не участвуют в размещении обычной нагрузки.

Если же на них сидят еще какие-то workload’ы, кроме ingress, то тогда лучше не исключать их целиком, а вводить более точную логику. Но по вашему описанию как раз нужно именно исключение.

Одна полезная мелочь, которую стоит добавить следующей: флаг вроде `--report-file report.json`, чтобы второй скрипт кроме stdout еще сохранял итоговый отчет в файл.
