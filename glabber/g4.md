Я бы сделал не просто набор команд, а полноценный smoke-test.

Он проверяет:

1. что агент видит UserParameter;
2. что discovery возвращает валидный JSON;
3. что discovery реально нашел объекты;
4. что по найденному объекту можно получить значение;
5. выводит результаты в понятном виде.

Например:

```bash
#!/bin/bash

set -euo pipefail

AGENTD="/usr/sbin/zabbix_agentd"

GREEN="\033[0;32m"
RED="\033[0;31m"
YELLOW="\033[1;33m"
NC="\033[0m"

ok() {
    echo -e "${GREEN}[ OK ]${NC} $*"
}

warn() {
    echo -e "${YELLOW}[WARN]${NC} $*"
}

fail() {
    echo -e "${RED}[FAIL]${NC} $*"
    exit 1
}

echo
echo "====================================="
echo " Kubernetes -> Prometheus -> Zabbix "
echo "====================================="
echo

#
# Проверка cluster metrics
#

echo "Проверка cluster metrics..."

for metric in \
    oomkilled \
    crashloop \
    pending \
    failed \
    deployment_not_ready \
    pvc_not_bound
do

    result="$($AGENTD -t "k8s.cluster.metric[$metric]" 2>/dev/null || true)"

    if [[ "$result" == *"Unsupported"* ]]; then
        fail "cluster metric $metric unsupported"
    fi

    ok "cluster metric $metric"
done

echo

#
# Проверка discovery нод
#

echo "Проверка node discovery..."

NODE_JSON="$($AGENTD -t k8s.node.discovery 2>/dev/null | sed 's/^.*|\(.*\)$/\1/')"

if ! echo "$NODE_JSON" | jq . >/dev/null 2>&1
then
    fail "node discovery вернул невалидный JSON"
fi

NODE_COUNT="$(echo "$NODE_JSON" | jq '.data | length')"

ok "найдено нод: $NODE_COUNT"

if [[ "$NODE_COUNT" -eq 0 ]]
then
    warn "ноды не найдены"
else

    FIRST_NODE="$(echo "$NODE_JSON" | jq -r '.data[0]."{#NODE}"')"

    echo "Тестируем ноду: $FIRST_NODE"

    for cond in \
        Ready \
        MemoryPressure \
        DiskPressure \
        PIDPressure
    do

        result="$(
            $AGENTD -t \
            "k8s.node.condition[$cond,$FIRST_NODE]" \
            2>/dev/null || true
        )"

        if [[ "$result" == *"Unsupported"* ]]
        then
            fail "$cond для $FIRST_NODE unsupported"
        fi

        ok "$cond для $FIRST_NODE"
    done
fi

echo

#
# Проверка discovery ingress
#

echo "Проверка ingress service/status discovery..."

INGRESS_JSON="$(
    $AGENTD -t k8s.ingress.service_status.discovery \
    2>/dev/null | sed 's/^.*|\(.*\)$/\1/'
)"

if ! echo "$INGRESS_JSON" | jq . >/dev/null 2>&1
then
    fail "ingress discovery вернул невалидный JSON"
fi

PAIR_COUNT="$(echo "$INGRESS_JSON" | jq '.data | length')"

ok "найдено service/status пар: $PAIR_COUNT"

if [[ "$PAIR_COUNT" -gt 0 ]]
then

    FIRST_SERVICE="$(
        echo "$INGRESS_JSON" |
        jq -r '.data[0]."{#SERVICE}"'
    )"

    FIRST_STATUS="$(
        echo "$INGRESS_JSON" |
        jq -r '.data[0]."{#STATUS}"'
    )"

    echo "Тестируем:"
    echo "  service=$FIRST_SERVICE"
    echo "  status=$FIRST_STATUS"

    result="$(
        $AGENTD -t \
        "k8s.ingress.status.service[$FIRST_STATUS,$FIRST_SERVICE]" \
        2>/dev/null || true
    )"

    if [[ "$result" == *"Unsupported"* ]]
    then
        fail "ingress.status.service unsupported"
    fi

    ok "ingress.status.service работает"
else
    warn "Ingress-данные отсутствуют"
fi

echo
ok "Все проверки завершены"
echo
```

Сохранить:

```bash
/usr/local/bin/test_k8s_prometheus_zabbix.sh
```

Права:

```bash
chmod +x /usr/local/bin/test_k8s_prometheus_zabbix.sh
```

Запуск:

```bash
/usr/local/bin/test_k8s_prometheus_zabbix.sh
```
