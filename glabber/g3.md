Ниже полный вариант с **универсальным кэшем всех Prometheus-запросов на 60 секунд** и комментариями.

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Единый скрипт для zabbix-agent / glabber.

Назначение:
  - получать метрики из Prometheus;
  - отдавать значения в формате, понятном zabbix-agent / glabber;
  - делать LLD discovery для нод и ingress service/status;
  - кэшировать ВСЕ запросы к Prometheus на CACHE_TTL секунд.

Важно:
  zabbix-agent запускает UserParameter каждый раз как отдельный процесс.
  Поэтому кэш хранится не в памяти, а в файлах на диске.
"""

import hashlib
import json
import os
import sys
import time
import urllib.parse
import urllib.request


# URL Prometheus.
# Можно переопределить через переменную окружения:
#   PROM_URL=http://prometheus.local:9090
PROM_URL = os.environ.get("PROM_URL", "http://127.0.0.1:9090")

# Таймаут HTTP-запроса к Prometheus.
TIMEOUT = int(os.environ.get("PROM_TIMEOUT", "10"))

# Время жизни кэша в секундах.
CACHE_TTL = int(os.environ.get("PROM_CACHE_TTL", "60"))

# Каталог для файлового кэша.
# Для production лучше /var/cache/k8s-prometheus,
# но /tmp проще для первого запуска.
CACHE_DIR = os.environ.get("PROM_CACHE_DIR", "/tmp/k8s-prometheus-cache")


CLUSTER_QUERIES = {
    "oomkilled": (
        'sum(kube_pod_container_status_last_terminated_reason{reason="OOMKilled"})'
    ),
    "crashloop": (
        'sum(kube_pod_container_status_waiting_reason{reason="CrashLoopBackOff"})'
    ),
    "pending": (
        'sum(kube_pod_status_phase{phase="Pending"})'
    ),
    "failed": (
        'sum(kube_pod_status_phase{phase="Failed"})'
    ),
    "deployment_not_ready": (
        "sum("
        "  clamp_min("
        "    kube_deployment_spec_replicas"
        "    -"
        "    kube_deployment_status_ready_replicas,"
        "    0"
        "  )"
        ")"
    ),
    "pvc_not_bound": (
        'sum(kube_persistentvolumeclaim_status_phase{phase!="Bound"})'
    ),
}


NODE_CONDITIONS = {
    "Ready": "Ready",
    "MemoryPressure": "MemoryPressure",
    "DiskPressure": "DiskPressure",
    "PIDPressure": "PIDPressure",
}


def ensure_cache_dir():
    """
    Создает каталог для кэша, если его еще нет.

    Пользователь, от которого работает zabbix-agent,
    должен иметь права на запись в этот каталог.
    """
    os.makedirs(CACHE_DIR, exist_ok=True)


def normalize_query(query):
    """
    Убирает лишние пробелы и переводы строк из PromQL.

    Это нужно, чтобы один и тот же запрос,
    записанный в несколько строк или в одну строку,
    попадал в один и тот же файл кэша.
    """
    return " ".join(query.split())


def cache_file_for_query(query):
    """
    Возвращает имя файла кэша для конкретного PromQL-запроса.

    В имя файла не кладем сам PromQL, потому что там могут быть
    кавычки, скобки, слэши и длинные строки.
    Вместо этого считаем SHA1 от нормализованного запроса.
    """
    normalized = normalize_query(query)
    cache_key = hashlib.sha1(normalized.encode("utf-8")).hexdigest()
    return os.path.join(CACHE_DIR, f"{cache_key}.json")


def read_cache(query):
    """
    Пытается прочитать свежий ответ Prometheus из файлового кэша.

    Возвращает:
      - result, если кэш есть и он не устарел;
      - None, если кэша нет, он битый или устарел.
    """
    try:
        cache_file = cache_file_for_query(query)

        if not os.path.exists(cache_file):
            return None

        with open(cache_file, "r", encoding="utf-8") as f:
            payload = json.load(f)

        timestamp = payload.get("timestamp", 0)

        if time.time() - timestamp > CACHE_TTL:
            return None

        return payload.get("result")

    except Exception:
        # Ошибка чтения кэша не должна ломать сбор метрик.
        # Просто считаем, что кэша нет.
        return None


def write_cache(query, result):
    """
    Сохраняет ответ Prometheus в файловый кэш.

    Запись делается через временный файл и os.replace(),
    чтобы снизить шанс прочитать частично записанный JSON,
    если несколько процессов zabbix-agent запустятся одновременно.
    """
    try:
        ensure_cache_dir()

        cache_file = cache_file_for_query(query)
        tmp_file = f"{cache_file}.{os.getpid()}.tmp"

        payload = {
            "timestamp": time.time(),
            "query": normalize_query(query),
            "result": result,
        }

        with open(tmp_file, "w", encoding="utf-8") as f:
            json.dump(payload, f, ensure_ascii=False)

        os.replace(tmp_file, cache_file)

    except Exception:
        # Ошибка записи кэша не должна мешать вернуть метрику.
        pass


def prometheus_query(query):
    """
    Выполняет instant query к Prometheus API /api/v1/query.

    Здесь реализован универсальный кэш:
      1. сначала ищем свежий ответ в файле;
      2. если есть — возвращаем его;
      3. если нет — идем в Prometheus;
      4. успешный ответ сохраняем в кэш.

    На выходе возвращается data.result из ответа Prometheus.
    """
    cached_result = read_cache(query)

    if cached_result is not None:
        return cached_result

    normalized_query = normalize_query(query)

    params = urllib.parse.urlencode({"query": normalized_query})
    url = f"{PROM_URL}/api/v1/query?{params}"

    with urllib.request.urlopen(url, timeout=TIMEOUT) as response:
        data = json.loads(response.read().decode("utf-8"))

    if data.get("status") != "success":
        raise RuntimeError(data)

    result = data.get("data", {}).get("result", [])

    write_cache(query, result)

    return result


def get_single_value(query):
    """
    Выполняет PromQL-запрос, который должен вернуть одно значение.

    Если Prometheus ничего не вернул, возвращаем 0.
    Это удобно для триггеров: нет серии — нет проблемы.
    """
    result = prometheus_query(query)

    if not result:
        return 0.0

    value = result[0].get("value", [None, 0])[1]

    try:
        return float(value)
    except Exception:
        return 0.0


def print_value(value):
    """
    Печатает число в удобном виде.

    Если значение целое — печатаем без .0.
    Например:
      3.0 -> 3
      3.25 -> 3.25
    """
    try:
        value = float(value)
    except Exception:
        print(0)
        return

    if int(value) == value:
        print(int(value))
    else:
        print(value)


def node_discovery():
    """
    LLD discovery нод Kubernetes.

    Возвращает JSON вида:
      {
        "data": [
          {"{#NODE}": "node1"},
          {"{#NODE}": "node2"}
        ]
      }
    """
    result = prometheus_query("kube_node_info")

    nodes = []
    seen = set()

    for item in result:
        node = item.get("metric", {}).get("node")

        if node and node not in seen:
            seen.add(node)
            nodes.append({"{#NODE}": node})

    print(json.dumps({"data": nodes}, ensure_ascii=False))


def ingress_service_status_discovery():
    """
    LLD discovery пар service/status для ingress.

    Используется, если в glabber нужно создать item prototype
    на каждую пару:
      service + HTTP status

    Запрос сразу группирует данные по service/status.
    Благодаря универсальному кэшу этот же PromQL потом может быть
    использован при получении конкретного значения.
    """
    query = """
        sum by (service, status) (
          increase(nginx_ingress_controller_request[1m])
        )
    """

    result = prometheus_query(query)

    items = []
    seen = set()

    for item in result:
        metric = item.get("metric", {})
        service = metric.get("service")
        status = metric.get("status")

        if not service or not status:
            continue

        key = (service, status)

        if key in seen:
            continue

        seen.add(key)

        items.append({
            "{#SERVICE}": service,
            "{#STATUS}": status,
        })

    print(json.dumps({"data": items}, ensure_ascii=False))


def cluster_metric(name):
    """
    Возвращает одну агрегированную метрику по кластеру.

    Примеры:
      k8s.cluster.metric[oomkilled]
      k8s.cluster.metric[crashloop]
      k8s.cluster.metric[pending]
    """
    query = CLUSTER_QUERIES[name]
    print_value(get_single_value(query))


def node_condition(condition, node):
    """
    Возвращает состояние конкретной condition на конкретной ноде.

    Для Ready:
      1 = нода Ready
      0 = нода NotReady

    Для Pressure-состояний:
      1 = pressure есть
      0 = pressure нет
    """
    query = (
        'kube_node_status_condition'
        f'{{node="{node}",condition="{condition}",status="true"}}'
    )

    print_value(get_single_value(query))


def ingress_status_service(status, service):
    """
    Возвращает количество запросов с конкретным HTTP status
    для конкретного service за последнюю минуту.

    Важно:
      здесь используется тот же общий PromQL, что и в discovery:
        sum by (service, status) (...)

      То есть Prometheus вызывается один раз,
      а затем конкретное значение выбирается из результата.
    """
    query = """
        sum by (service, status) (
          increase(nginx_ingress_controller_request[1m])
        )
    """

    result = prometheus_query(query)

    for item in result:
        metric = item.get("metric", {})

        if metric.get("service") != service:
            continue

        if metric.get("status") != status:
            continue

        value = item.get("value", [None, 0])[1]
        print_value(value)
        return

    print(0)


def usage():
    """
    Печатает справку по использованию скрипта.
    """
    print("""
Usage:

  k8s_prometheus.py node.discovery
  k8s_prometheus.py node.condition <condition> <node>

  k8s_prometheus.py cluster <metric>

  k8s_prometheus.py ingress.service_status.discovery
  k8s_prometheus.py ingress.status.service <status> <service>

Cluster metrics:
  oomkilled
  crashloop
  pending
  failed
  deployment_not_ready
  pvc_not_bound

Node conditions:
  Ready
  MemoryPressure
  DiskPressure
  PIDPressure

Environment variables:
  PROM_URL
  PROM_TIMEOUT
  PROM_CACHE_TTL
  PROM_CACHE_DIR
""".strip())


def main():
    """
    Точка входа.

    Важно для zabbix-agent:
      при любой ошибке печатаем 0 и завершаемся с кодом 0.
      Иначе item может стать unsupported.
    """
    try:
        if len(sys.argv) < 2:
            usage()
            sys.exit(1)

        command = sys.argv[1]

        if command == "node.discovery":
            node_discovery()
            return

        if command == "node.condition":
            if len(sys.argv) != 4:
                usage()
                sys.exit(1)

            condition = sys.argv[2]
            node = sys.argv[3]

            if condition not in NODE_CONDITIONS:
                print(0)
                return

            node_condition(condition, node)
            return

        if command == "cluster":
            if len(sys.argv) != 3:
                usage()
                sys.exit(1)

            metric = sys.argv[2]

            if metric not in CLUSTER_QUERIES:
                print(0)
                return

            cluster_metric(metric)
            return

        if command == "ingress.service_status.discovery":
            ingress_service_status_discovery()
            return

        if command == "ingress.status.service":
            if len(sys.argv) != 4:
                usage()
                sys.exit(1)

            status = sys.argv[2]
            service = sys.argv[3]

            ingress_status_service(status, service)
            return

        usage()
        sys.exit(1)

    except Exception:
        print(0)
        sys.exit(0)


if __name__ == "__main__":
    main()
```

Конфиг агента остается таким:

```ini
UserParameter=k8s.node.discovery,/etc/zabbix/scripts/k8s_prometheus.py node.discovery
UserParameter=k8s.node.condition[*],/etc/zabbix/scripts/k8s_prometheus.py node.condition "$1" "$2"

UserParameter=k8s.cluster.metric[*],/etc/zabbix/scripts/k8s_prometheus.py cluster "$1"

UserParameter=k8s.ingress.service_status.discovery,/etc/zabbix/scripts/k8s_prometheus.py ingress.service_status.discovery
UserParameter=k8s.ingress.status.service[*],/etc/zabbix/scripts/k8s_prometheus.py ingress.status.service "$1" "$2"
```

Для `/var/cache` лучше так:

```bash
mkdir -p /var/cache/k8s-prometheus
chown zabbix:zabbix /var/cache/k8s-prometheus
```

и в systemd/env или в скрипте указать:

```bash
PROM_CACHE_DIR=/var/cache/k8s-prometheus
```
