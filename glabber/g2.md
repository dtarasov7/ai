Да. Ниже единый Python-скрипт для всех обсужденных метрик, включая первый кейс: **HTTP status code по ingress в разрезе service**.

`/etc/zabbix/scripts/k8s_prometheus.py`

```python
#!/usr/bin/env python3

import json
import sys
import urllib.parse
import urllib.request

PROM_URL = "http://127.0.0.1:9090"
TIMEOUT = 10


CLUSTER_QUERIES = {
    "oomkilled": 'sum(kube_pod_container_status_last_terminated_reason{reason="OOMKilled"})',
    "crashloop": 'sum(kube_pod_container_status_waiting_reason{reason="CrashLoopBackOff"})',
    "pending": 'sum(kube_pod_status_phase{phase="Pending"})',
    "failed": 'sum(kube_pod_status_phase{phase="Failed"})',
    "deployment_not_ready": '''
        sum(
          clamp_min(
            kube_deployment_spec_replicas
            -
            kube_deployment_status_ready_replicas,
            0
          )
        )
    ''',
    "pvc_not_bound": 'sum(kube_persistentvolumeclaim_status_phase{phase!="Bound"})',
}


NODE_CONDITIONS = {
    "Ready": "Ready",
    "MemoryPressure": "MemoryPressure",
    "DiskPressure": "DiskPressure",
    "PIDPressure": "PIDPressure",
}


def prometheus_query(query):
    params = urllib.parse.urlencode({"query": query})
    url = f"{PROM_URL}/api/v1/query?{params}"

    with urllib.request.urlopen(url, timeout=TIMEOUT) as response:
        data = json.loads(response.read().decode("utf-8"))

    if data.get("status") != "success":
        raise RuntimeError(data)

    return data.get("data", {}).get("result", [])


def get_single_value(query):
    result = prometheus_query(query)

    if not result:
        return 0

    value = result[0].get("value", [None, 0])[1]

    try:
        return float(value)
    except Exception:
        return 0


def print_value(value):
    if int(value) == value:
        print(int(value))
    else:
        print(value)


def node_discovery():
    query = 'kube_node_info'
    result = prometheus_query(query)

    nodes = []
    seen = set()

    for item in result:
        node = item.get("metric", {}).get("node")
        if node and node not in seen:
            seen.add(node)
            nodes.append({"{#NODE}": node})

    print(json.dumps({"data": nodes}, ensure_ascii=False))


def ingress_service_discovery():
    query = 'count by (service) (nginx_ingress_controller_request)'
    result = prometheus_query(query)

    services = []
    seen = set()

    for item in result:
        service = item.get("metric", {}).get("service")
        if service and service not in seen:
            seen.add(service)
            services.append({"{#SERVICE}": service})

    print(json.dumps({"data": services}, ensure_ascii=False))


def cluster_metric(name):
    query = CLUSTER_QUERIES[name]
    print_value(get_single_value(query))


def node_condition(condition, node):
    query = (
        'kube_node_status_condition'
        f'{{node="{node}",condition="{condition}",status="true"}}'
    )
    print_value(get_single_value(query))


def ingress_status_service(status, service):
    query = (
        'sum(increase('
        'nginx_ingress_controller_request'
        f'{{status="{status}",service="{service}"}}[1m]'
        '))'
    )
    print_value(get_single_value(query))


def usage():
    print("""
Usage:

  k8s_prometheus.py node.discovery
  k8s_prometheus.py node.condition <condition> <node>

  k8s_prometheus.py cluster <metric>

  k8s_prometheus.py ingress.service.discovery
  k8s_prometheus.py ingress.status.service <status> <service>

Cluster metrics:
  oomkilled
  crashloop
  pending
  failed
  deployment_not_ready
  pvc_not_bound

Node conditions:
  Ready
  MemoryPressure
  DiskPressure
  PIDPressure
""".strip())


def main():
    try:
        if len(sys.argv) < 2:
            usage()
            sys.exit(1)

        command = sys.argv[1]

        if command == "node.discovery":
            node_discovery()
            return

        if command == "node.condition":
            if len(sys.argv) != 4:
                usage()
                sys.exit(1)

            condition = sys.argv[2]
            node = sys.argv[3]

            if condition not in NODE_CONDITIONS:
                print(0)
                return

            node_condition(condition, node)
            return

        if command == "cluster":
            if len(sys.argv) != 3:
                usage()
                sys.exit(1)

            metric = sys.argv[2]

            if metric not in CLUSTER_QUERIES:
                print(0)
                return

            cluster_metric(metric)
            return

        if command == "ingress.service.discovery":
            ingress_service_discovery()
            return

        if command == "ingress.status.service":
            if len(sys.argv) != 4:
                usage()
                sys.exit(1)

            status = sys.argv[2]
            service = sys.argv[3]
            ingress_status_service(status, service)
            return

        usage()
        sys.exit(1)

    except Exception:
        print(0)
        sys.exit(0)


if __name__ == "__main__":
    main()
```

Права:

```bash
chmod +x /etc/zabbix/scripts/k8s_prometheus.py
chown zabbix:zabbix /etc/zabbix/scripts/k8s_prometheus.py
```

Конфиг `zabbix-agentd`:

```ini
UserParameter=k8s.node.discovery,/etc/zabbix/scripts/k8s_prometheus.py node.discovery
UserParameter=k8s.node.condition[*],/etc/zabbix/scripts/k8s_prometheus.py node.condition "$1" "$2"

UserParameter=k8s.cluster.metric[*],/etc/zabbix/scripts/k8s_prometheus.py cluster "$1"

UserParameter=k8s.ingress.service.discovery,/etc/zabbix/scripts/k8s_prometheus.py ingress.service.discovery
UserParameter=k8s.ingress.status.service[*],/etc/zabbix/scripts/k8s_prometheus.py ingress.status.service "$1" "$2"
```
