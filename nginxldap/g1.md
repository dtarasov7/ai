Да, вот более практичный вариант: **без uvicorn**, на `ThreadingHTTPServer`, с LDAP bind, TTL-кэшем и логированием неуспешных LDAP-проверок.

Пароли и `Authorization` в лог **не пишутся**.

### `auth_service.py`

```python id="w6k2ch"
#!/usr/bin/env python3

import base64
import hashlib
import hmac
import logging
import os
import secrets
import time
from dataclasses import dataclass
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer
from typing import Optional
from urllib.parse import urlparse

from ldap3 import Server, Connection
from ldap3.core.exceptions import LDAPException, LDAPSocketOpenError


# =========================
# Configuration
# =========================

AUTH_HOST = os.getenv("AUTH_HOST", "127.0.0.1")
AUTH_PORT = int(os.getenv("AUTH_PORT", "8888"))

LDAP_SERVER = os.getenv("LDAP_SERVER", "ldaps://dc01.example.com:636")

# Для AD обычно удобно:
#   {username}@example.com
# или:
#   EXAMPLE\{username}
LDAP_USER_TEMPLATE = os.getenv("LDAP_USER_TEMPLATE", "{username}@example.com")

LDAP_CONNECT_TIMEOUT = int(os.getenv("LDAP_CONNECT_TIMEOUT", "5"))
LDAP_RECEIVE_TIMEOUT = int(os.getenv("LDAP_RECEIVE_TIMEOUT", "5"))

# Кэш успешных и неуспешных проверок.
# Успешные можно кэшировать дольше, неуспешные — очень коротко.
CACHE_OK_TTL = int(os.getenv("CACHE_OK_TTL", "60"))
CACHE_FAIL_TTL = int(os.getenv("CACHE_FAIL_TTL", "5"))

# Секрет для HMAC-ключа кэша.
# Если не задан — генерируется при старте процесса.
# После рестарта кэш станет пустым, это нормально.
CACHE_SECRET = os.getenv("AUTH_CACHE_SECRET")
if not CACHE_SECRET:
    CACHE_SECRET = secrets.token_hex(32)

LOG_LEVEL = os.getenv("LOG_LEVEL", "INFO").upper()


# =========================
# Logging
# =========================

logging.basicConfig(
    level=LOG_LEVEL,
    format="%(asctime)s %(levelname)s %(threadName)s %(message)s",
)

logger = logging.getLogger("ldap-auth-service")


# =========================
# Cache
# =========================

@dataclass
class CacheEntry:
    allowed: bool
    expires_at: float


auth_cache: dict[str, CacheEntry] = {}


def now() -> float:
    return time.time()


def safe_username(username: str) -> str:
    """
    Защита от мусора в логах: убираем переносы строк и ограничиваем длину.
    """
    username = username.replace("\r", "_").replace("\n", "_")
    return username[:128]


def make_cache_key(username: str, password: str) -> str:
    """
    Не храним пароль в кэше.
    Ключ — HMAC(username + password).
    """
    raw = f"{username}\0{password}".encode("utf-8", errors="replace")
    return hmac.new(
        CACHE_SECRET.encode("utf-8"),
        raw,
        hashlib.sha256,
    ).hexdigest()


def cache_get(username: str, password: str) -> Optional[bool]:
    key = make_cache_key(username, password)
    entry = auth_cache.get(key)

    if not entry:
        return None

    if entry.expires_at < now():
        auth_cache.pop(key, None)
        return None

    return entry.allowed


def cache_set(username: str, password: str, allowed: bool) -> None:
    key = make_cache_key(username, password)

    ttl = CACHE_OK_TTL if allowed else CACHE_FAIL_TTL

    if ttl <= 0:
        return

    auth_cache[key] = CacheEntry(
        allowed=allowed,
        expires_at=now() + ttl,
    )


def cache_cleanup() -> None:
    """
    Простая lazy-очистка устаревших записей.
    """
    current = now()
    expired_keys = [
        key for key, entry in auth_cache.items()
        if entry.expires_at < current
    ]

    for key in expired_keys:
        auth_cache.pop(key, None)


# =========================
# Basic Auth parsing
# =========================

def parse_basic_auth(header: Optional[str]) -> Optional[tuple[str, str]]:
    if not header:
        return None

    if not header.lower().startswith("basic "):
        return None

    encoded = header.split(" ", 1)[1].strip()

    try:
        decoded = base64.b64decode(encoded, validate=True).decode("utf-8")
    except Exception:
        return None

    if ":" not in decoded:
        return None

    username, password = decoded.split(":", 1)

    if not username or not password:
        return None

    return username, password


# =========================
# LDAP check
# =========================

def check_ldap_login(username: str, password: str, client_ip: str) -> bool:
    """
    Проверяет логин/пароль через LDAP bind.

    True  — bind успешен
    False — AD отверг логин/пароль
    Exception — проблема с LDAP-сервером, сетью, TLS и т.д.
    """
    ldap_user = LDAP_USER_TEMPLATE.format(username=username)

    server = Server(
        LDAP_SERVER,
        connect_timeout=LDAP_CONNECT_TIMEOUT,
    )

    conn = None

    try:
        conn = Connection(
            server,
            user=ldap_user,
            password=password,
            receive_timeout=LDAP_RECEIVE_TIMEOUT,
            auto_bind=False,
        )

        bind_ok = conn.bind()

        if bind_ok:
            return True

        result = conn.result or {}
        result_code = result.get("result")
        description = result.get("description")
        message = result.get("message")

        logger.warning(
            "LDAP password check failed: username=%s client_ip=%s "
            "ldap_user=%s result=%s description=%s message=%s",
            safe_username(username),
            client_ip,
            safe_username(ldap_user),
            result_code,
            description,
            message,
        )

        return False

    finally:
        if conn is not None and conn.bound:
            conn.unbind()


# =========================
# HTTP handler
# =========================

class AuthHandler(BaseHTTPRequestHandler):
    server_version = "LDAPAuthService/1.0"

    def do_GET(self):
        parsed_path = urlparse(self.path)

        if parsed_path.path == "/health":
            self.send_json(200, '{"status":"ok"}')
            return

        if parsed_path.path != "/auth":
            self.send_response(404)
            self.end_headers()
            return

        self.handle_auth()

    def do_HEAD(self):
        parsed_path = urlparse(self.path)

        if parsed_path.path == "/auth":
            self.handle_auth(head_only=True)
            return

        self.send_response(404)
        self.end_headers()

    def handle_auth(self, head_only: bool = False):
        client_ip = self.get_client_ip()

        credentials = parse_basic_auth(self.headers.get("Authorization"))

        if not credentials:
            self.send_response(401)
            self.send_header("WWW-Authenticate", 'Basic realm="AD Login"')
            self.end_headers()
            return

        username, password = credentials

        cached = cache_get(username, password)

        if cached is not None:
            if cached:
                self.send_response(204)
                self.end_headers()
                return

            logger.warning(
                "Auth rejected from cache: username=%s client_ip=%s",
                safe_username(username),
                client_ip,
            )

            self.send_response(401)
            self.end_headers()
            return

        try:
            allowed = check_ldap_login(username, password, client_ip)

            cache_set(username, password, allowed)
            cache_cleanup()

            if allowed:
                self.send_response(204)
                self.end_headers()
                return

            self.send_response(401)
            self.end_headers()
            return

        except LDAPSocketOpenError as exc:
            logger.error(
                "LDAP connection error: username=%s client_ip=%s error=%s",
                safe_username(username),
                client_ip,
                str(exc),
            )
            self.send_response(500)
            self.end_headers()
            return

        except LDAPException as exc:
            logger.error(
                "LDAP error: username=%s client_ip=%s error=%s",
                safe_username(username),
                client_ip,
                str(exc),
            )
            self.send_response(500)
            self.end_headers()
            return

        except Exception as exc:
            logger.exception(
                "Unexpected auth-service error: username=%s client_ip=%s error=%s",
                safe_username(username),
                client_ip,
                str(exc),
            )
            self.send_response(500)
            self.end_headers()
            return

    def get_client_ip(self) -> str:
        """
        nginx может передавать реальный IP пользователя через X-Real-IP.
        Если заголовка нет, берём адрес TCP-клиента.
        """
        x_real_ip = self.headers.get("X-Real-IP")

        if x_real_ip:
            return x_real_ip.strip()[:64]

        return self.client_address[0]

    def send_json(self, status_code: int, body: str):
        body_bytes = body.encode("utf-8")

        self.send_response(status_code)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(body_bytes)))
        self.end_headers()
        self.wfile.write(body_bytes)

    def log_message(self, fmt, *args):
        """
        Отключаем стандартный access-log http.server,
        чтобы случайно не логировать лишние заголовки.
        """
        return


# =========================
# Main
# =========================

def main():
    logger.info(
        "Starting LDAP auth service on %s:%s, LDAP_SERVER=%s, LDAP_USER_TEMPLATE=%s, "
        "CACHE_OK_TTL=%ss, CACHE_FAIL_TTL=%ss",
        AUTH_HOST,
        AUTH_PORT,
        LDAP_SERVER,
        LDAP_USER_TEMPLATE,
        CACHE_OK_TTL,
        CACHE_FAIL_TTL,
    )

    httpd = ThreadingHTTPServer((AUTH_HOST, AUTH_PORT), AuthHandler)

    try:
        httpd.serve_forever()
    except KeyboardInterrupt:
        logger.info("Stopping LDAP auth service")
    finally:
        httpd.server_close()


if __name__ == "__main__":
    main()
```

### Настройка кэша

По умолчанию:

```bash id="n68ihs"
CACHE_OK_TTL=60
CACHE_FAIL_TTL=5
```

То есть:

```text id="tci78t"
успешная авторизация  → кэшируется на 60 секунд
неуспешная авторизация → кэшируется на 5 секунд
```

Можно изменить:

```bash id="lqu9np"
export CACHE_OK_TTL=300
export CACHE_FAIL_TTL=10
```

Или полностью выключить кэш:

```bash id="aeq8le"
export CACHE_OK_TTL=0
export CACHE_FAIL_TTL=0
```

    internal;

}

### Проверка

```bash id="icg7l8"
curl -i -u 'user:password' http://127.0.0.1:8888/auth
```

Успешный ответ:

```text id="n9w4fc"
HTTP/1.0 204 No Content
```

Неверный пароль:

```text id="1lfexa"
HTTP/1.0 401 Unauthorized
```

В логе при неправильном пароле будет примерно так:

```text id="c9gnzj"
WARNING LDAP password check failed: username=user client_ip=127.0.0.1 ldap_user=user@example.com result=49 description=invalidCredentials message=...
```

Главное: **пароль не логируется**, `Authorization` не логируется, а ключ кэша строится через HMAC, поэтому в памяти нет обычного `username:password` в качестве ключа.
